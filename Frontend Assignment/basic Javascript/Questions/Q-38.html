2) Alphabetic Pattern:

        <script>
        // Alphabetic Pattern
        function printAlphabeticPattern(rows) {
            let currentChar = 65; // ASCII code for 'A'
            for (let i = 1; i <= rows; i++) {
                for (let j = 1; j <= i; j++) {
                    document.write(`${String.fromCharCode(currentChar)} `);
                    currentChar++;
                }
                document.write("< br>");
            }
        }

        // Example usage
        document.write("< strong>2) Alphabetic Pattern:< /strong>< br>");
        printAlphabeticPattern(5);
        document.write("< br>");
        </script>
