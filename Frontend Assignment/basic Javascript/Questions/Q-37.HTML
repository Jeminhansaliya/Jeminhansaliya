Q.37 Use console.log() and escape characters to print the following pattern in JS?
1 1 1 1 1
2 1 2 4 8
3 1 3 9 27
4 1 4 16 64
5 1 5 25 125

Program :


                <script>
                function printPattern(rows) 
                {
                    for (let i = 1; i <= rows; i++) 
                    {
                        let row = '';
                        for (let j = 1; j <= 5; j++) 
                        {
                            if (j === 1) {
                                row += `${i} `;
                            } else {
                                row += `${Math.pow(i, j)} `;
                            } 
                        }
                        document.write("<br>"+"\n"+row);       
                    }
                }
                // printPattern(5);
                document.getElementById("ans").innerHTML=printPattern(5);
                </script>
