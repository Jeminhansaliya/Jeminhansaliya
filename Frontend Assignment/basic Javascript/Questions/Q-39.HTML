Q.39 Accept 3 numbers from user using while loop and check each numbers palindrome?

Program :


        <script>
                // Function to check if a number is a palindrome
        function isPalindrome(number) {
            const originalString = String(number);
            const reversedString = originalString.split('').reverse().join('');
            return originalString === reversedString;
        }

        // Accept 3 numbers from the user and check if each is a palindrome
        let count = 1;
        
            const userNumber = 303;
        while (count <= 1) {
            

            // Check if the entered number is a palindrome
            if (!isNaN(userNumber)) {
                if (isPalindrome(userNumber)) {
                    document.write(`${userNumber} is a palindrome.`);
                } else {
                    document.write(`${userNumber} is not a palindrome.`);
                }
                count++;
            } else {
                document.write("Invalid input. Please enter a valid number.");
            }
        }
        </script>
