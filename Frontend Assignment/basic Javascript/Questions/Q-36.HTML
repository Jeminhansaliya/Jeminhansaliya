36. Write a program you have to make a summation of first and last Digit. (E.g., 1234 Ans: - 5) in JS?

Program :

    
                <h3 id="ans"></h3>
    
                <script>
                let num = prompt("Enter Four Digits Number:")
                function sumFirstLastDigits(num){
                    const numStr = String(num);
                    const firstDigit = Number(numStr[0]);
                    const LastDigit = Number(numStr[numStr.length-1]);
                    return firstDigit + LastDigit;
                }
                // document.write(sumFirstLastDigits(num));
                document.getElementById("ans").innerHTML=sumFirstLastDigits(num);
                </script>
