Q.65 Find the length of a string without using libraryFunction?

Program :

    <script>
    function findStringLength(inputString) {
        let length = 0;

        // Iterate through each character in the string
        for (let char of inputString) {
            length++;
        }

        return length;
    }

    // Example usage
    let myStr = "Hello, World!";
    let lengthWithoutLibraryFunction = findStringLength(myStr);

    document.write("Length without using library function:" + lengthWithoutLibraryFunction);
    </script>
