Program :


                <h3 id="ans"></h3>
    
                <script>
            
                    let input = prompt("Enter the number to get factorial of: ");
                    let text = input;
                    for (let i = 1; i < input; i++) {
                    text = i * text;
                    }
            
                    document.getElementById("ans").innerHTML = text;
                </script>
