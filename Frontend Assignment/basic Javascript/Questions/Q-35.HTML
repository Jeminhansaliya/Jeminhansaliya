35. Write a program make a summation of given number (E.g., 1523 Ans: - 11) in JS?

Program :


            <h3 id="ans"></h3>

            <script>
            let n = prompt ("Enter Four Digits Number:");
            function getsum(n){
                let sum;
                for(sum = 0; n > 0; sum+= n % 10,n= parseInt(n/10));
                return sum;
            }
            
            document.getElementById("ans").innerHTML=getsum(n);
            </script>
