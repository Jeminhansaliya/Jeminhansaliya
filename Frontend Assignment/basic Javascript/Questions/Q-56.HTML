Q.56 Dynamic Form Validation in JS?

Program :

    
        <script>
        function validateEmail() {
            var emailInput = document.getElementById('email');
            var errorMessage = document.getElementById('error-message');

            // Simple email validation
            var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!emailRegex.test(emailInput.value)) {
                emailInput.classList.add('error');
                errorMessage.textContent = 'Invalid email address';
            } else {
                emailInput.classList.remove('error');
                errorMessage.textContent = '';
            }
        }

        function validateNumber() {
            var numberInput = document.getElementById('number');
            var errorMessage = document.getElementById('error-message');

            // Simple number validation
            if (isNaN(numberInput.value) || numberInput.value <= 0) {
                numberInput.classList.add('error');
                errorMessage.textContent = 'Invalid number';
            } else {
                numberInput.classList.remove('error');
                errorMessage.textContent = '';
            }
        }

        function validatePassword() {
            var passwordInput = document.getElementById('password');
            var errorMessage = document.getElementById('error-message');

            // Simple password length validation
            if (passwordInput.value.length < 8) {
                passwordInput.classList.add('error');
                errorMessage.textContent = 'Password must be at least 8 characters';
            } else {
                passwordInput.classList.remove('error');
                errorMessage.textContent = '';
            }
        }

        function validateForm() {
            // Run all individual field validations
            validateEmail();
            validateNumber();
            validatePassword();

            // Additional logic for overall form validation or submission
            // For now, you can perform form submission if no errors are present
        }
        </script>
